# Quick Reference Commands

## Local Development

# Install all dependencies
npm install
cd client && npm install && cd ..

# Start development server (both frontend and backend)
npm run dev

# Start backend only
npm run server

# Start frontend only
npm run client

# Build for production
npm run build

## Heroku Deployment

# Login to Heroku
heroku login

# Create new app
heroku create your-app-name

# Or create with random name
heroku create

# Set environment variables
heroku config:set NODE_ENV=production
heroku config:set JWT_SECRET=$(node -e "console.log(require('crypto').randomBytes(32).toString('hex'))")
heroku config:set JWT_REFRESH_SECRET=$(node -e "console.log(require('crypto').randomBytes(32).toString('hex'))")
heroku config:set CLIENT_URL=https://your-app-name.herokuapp.com

# Add MongoDB (Option 1: Heroku add-on)
heroku addons:create mongolab:sandbox

# Add MongoDB (Option 2: Atlas connection string)
heroku config:set MONGODB_URI="your_mongodb_connection_string"

# Initialize git (if not done)
git init
git add .
git commit -m "Initial commit"

# Add Heroku remote
heroku git:remote -a your-app-name

# Deploy to Heroku
git push heroku main

# Or if using master branch
git push heroku master

# Scale dyno
heroku ps:scale web=1

# Open app in browser
heroku open

# View logs
heroku logs --tail

# Restart app
heroku restart

# View environment variables
heroku config

# View app info
heroku apps:info

## MongoDB Setup

# Generate connection string format
# MongoDB Atlas: mongodb+srv://username:password@cluster.mongodb.net/dbname?retryWrites=true&w=majority

# For MongoDB Atlas:
# 1. Create cluster at https://cloud.mongodb.com
# 2. Create database user
# 3. Whitelist IP: 0.0.0.0/0 (for Heroku)
# 4. Get connection string
# 5. Replace <password> with actual password
# 6. Set in Heroku:
heroku config:set MONGODB_URI="your_connection_string"

## Local Testing

# Test if MongoDB is running
mongo --version

# Start MongoDB (Windows)
net start MongoDB

# Start MongoDB (Mac/Linux)
sudo systemctl start mongod

# Check port usage (Windows)
netstat -ano | findstr :5000
netstat -ano | findstr :3000

# Kill process on port (Windows)
taskkill /PID <process_id> /F

# Check port usage (Mac/Linux)
lsof -i :5000
lsof -i :3000

# Kill process on port (Mac/Linux)
kill -9 <process_id>

npm install --save-dev package-name

## Database Operations

# Connect to MongoDB (local)
mongo

# Show databases
show dbs

# Use specific database
use rental-platform

# Show collections
show collections

# Query collection
db.users.find()

# Drop collection
db.users.drop()

# Drop database
db.dropDatabase()
